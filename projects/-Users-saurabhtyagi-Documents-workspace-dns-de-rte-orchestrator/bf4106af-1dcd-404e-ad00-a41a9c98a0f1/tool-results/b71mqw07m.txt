+class S3AuditLogWriter:
+    """Writes a materialised audit log for a single erasure id to S3.
+
+    Produces the identical gzipped JSON Lines format that ``S3AuditLogReader``
+    consumes, so ``FallbackAuditLogReader`` continues to work without changes.
+    """
+
+    def __init__(self, bucket: str, prefix: str, s3_client: S3Client) -> None:
+        self.bucket: str = bucket
+        self.prefix: str = prefix.strip("/")
+        self.s3: S3Client = s3_client
+
+    def write(self, id: str, entries: list[AuditEntry]) -> None:
+        key = _s3_key_for(self.prefix, id)
+        lines = "\n".join(json.dumps(entry.to_json()) for entry in entries)
+        body = gzip.compress(lines.encode("utf-8"))
+        _ = self.s3.put_object(Bucket=self.bucket, Key=key, Body=body)
+        logger.info("Archived %d audit log entries to s3://%s/%s", len(entries), self.bucket, key)

+class AuditArchiver:
+    """Archives audit logs for an erasure request to S3 then removes them from DynamoDB.
+
+    Write order is intentional: S3 first, then DynamoDB delete. If the delete
+    fails mid-way the DynamoDB copy still exists so ``FallbackAuditLogReader``
+    will never return empty results. Retrying ``archive`` is safe — the S3
+    object is overwritten and ``delete_logs`` tolerates already-deleted items.
+    """
+
+    def __init__(
+        self, primary: DynamoDBAuditLogger, writer: S3AuditLogWriter
+    ) -> None:
+        self.primary: DynamoDBAuditLogger = primary
+        self.writer: S3AuditLogWriter = writer
+
+    def archive(self, id: str) -> None:
+        entries = self.primary.get_logs(id)
+        if not entries:
+            logger.info("No audit log entries to archive for id %s", id)
+            return
+        self.writer.write(id, entries)
+        self.primary.delete_logs(id)
+        logger.info("Audit log for id %s archived and removed from DynamoDB", id)
